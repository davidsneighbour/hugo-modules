{{- $path := . -}} {{- $type := "resource" -}} {{- $protocol := strings.Contains $path "://" -}} {{- if eq "true" $protocol -}} {{/* if path starts with http(s), then it's external */}} {{- $type := "external" -}} {{- else if eq "/" (substr $path 0 1) -}} {{/* if path starts with /, then it's inside the static directory */}} {{- $type := "static" -}} {{- end -}} {{/* else it's a resource */}}
