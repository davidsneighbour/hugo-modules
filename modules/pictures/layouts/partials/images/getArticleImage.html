{{/* Returns the article image for the given context as a resource object or an empty dict */}} {{/* Path of checking - resource named `*header*` - resource on a path `**\/header.*` - if resources exist then return the first image - placeholder image */}} {{- $context := . -}} {{- $articleimage := dict -}} {{- with $context.Resources.GetMatch "**/header*" -}} {{- $articleimage = . -}} {{- else -}} {{- $articleimages := $context.Resources.ByType "image" -}} {{- range first 1 $articleimages -}} {{- $articleimage = . -}} {{- end -}} {{- end -}} {{- with $articleimage -}}{{- else -}} {{- $config := partialCached "func/pictures/getConfiguration.html" . -}} {{- $default_image := $config.default_image | default "images/default.png" -}} {{- $articleimage = resources.Get $default_image -}} {{- end -}} {{- return $articleimage -}}
